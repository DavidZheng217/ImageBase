def generate_header(raw_head):
    '''
    Sample input:
    Accept: */*
    Accept-Encoding: br, gzip, deflate
    Accept-Language: zh-cn
    '''
    head = {}
    for line in raw_head.split('\n'):
        key, value = line.split(sep=':', maxsplit=1)
        head[key] = value
    return head

def pickNum(string, method):
    """
    Sample input: 
    string = ": \'([0-9+]) – ([0-9]+)"
    
    Sample Output:
    ('1', '2')
    """
    if (method == 'baths') | (method == 'beds'):
        numbers = re.findall('([0-9]+) – ([0-9]+)', string)
        if bool(numbers) == 0:
            numbers = re.findall('([0-9]+)', string)
    elif (method == 'price'):
        numbers = re.findall('\$(.+) – \$(.+)\+', string)
        if bool(numbers) == 0:
            numbers = re.findall('\$(.+)\+', string)
            if bool(numbers) == 0:
                numbers = re.findall('\$(.+)', string)
    elif (method == 'sqft'):
        numbers = re.findall('(.+) – (.+) sqft', string)
        if bool(numbers) == 0:
            numbers = re.findall('(.+) sqft', string)
    return numbers[0]

def string2float(stringList, avg=0):
    if len(stringList) > 2:
        return stringList.replace(',','')
    if avg:
        return np.mean([float(string.replace(',', '')) for string in stringList])
    else:
        return [float(string.replace(',', '')) for string in stringList]

def writeWithSep(lst, fhandle):
    '''
    lst is a list or tuple contains data for one observation
    '''
    for thing in lst:
        fhandle.write(str(thing))
        if thing == lst[-1]:

            fhandle.write(',')
